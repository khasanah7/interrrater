Is the utterance comprehensible?

# Add a column to identify the raters
Task1_CR1 <- Task1_CR1 %>%

# 2. Pivot: one row per rating
ratings_longCR1 <- Task1_CR1 %>%
  pivot_longer(
    cols = -Rater,
    names_to = "Item",
    values_to = "Score"
  )

# 3. Pivot: rows = items, columns = raters
ratings_wideCR1 <- ratings_longCR1 %>%
  pivot_wider(
    names_from = Rater,
    values_from = Score
  )

# 4. Make sure all values are numeric
ratings_wideCR1 <- ratings_wideCR1 %>%
  mutate(across(starts_with("R"), as.numeric))

# 5. Drop the Item column for ICC (just use numeric matrix)
icc_dataCR1 <- ratings_wideCR1 %>% select(-Item)

# 6. Compute ICC
icc(icc_dataCR1, model = "twoway", type = "consistency", unit = "average")

get_icc_summaryCR1 <- function(df, rater1, rater2) {
    r1_name <- rlang::as_label(rlang::ensym(rater1))
    r2_name <- rlang::as_label(rlang::ensym(rater2))
  
    result <- icc(df %>% select({{ rater1 }}, {{ rater2 }}), 
                model = "twoway", type = "consistency", unit = "average")
  
  tibble( 
    Raters = paste0(r1_name, " & ", r2_name),
    ICC = round(result$value, 3),
    CI_Lower = round(result$lbound, 3),
    CI_Upper = round(result$ubound, 3),
    P_Value = signif(result$p.value, 3)
  )
}

icc_pairwise_summaryCR1 <- bind_rows(
  get_icc_summaryCR1(icc_dataCR1, R1, R2),
  get_icc_summaryCR1(icc_dataCR1, R2, R3),
  get_icc_summaryCR1(icc_dataCR1, R1, R3)
)
